```jsx
import { useState, useEffect } from 'react';
import { useParams } from 'react-router-dom';
import axios from 'axios';
import { Avatar, Icon } from 'vant';
import 'github-markdown-css/github-markdown-light.css';

const ArticleDetail = () => {
  const { id } = useParams();
  const [article, setArticle] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);
  const [comments, setComments] = useState([]);
  const [inputComment, setInputComment] = useState('');
  const [isComment, setIsComment] = useState(false);

  useEffect(() => {
    fetchArticle();
  }, [id]);

  const fetchArticle = async () => {
    try {
      setLoading(true);
      const response = await axios.get(`/api/article/${id}`);
      setArticle(response.data);
      if (response.data.comments) {
        setComments(response.data.comments);
      }
      setError(null);
    } catch (err) {
      setError('获取文章失败');
      console.error('Error fetching article:', err);
    } finally {
      setLoading(false);
    }
  };

  const uploadComment = async () => {
    if (!inputComment.trim()) return;

    try {
      const response = await axios.post('/api/comment', {
        content: inputComment,
        aid: id
      });
      
      setComments([...comments, response.data]);
      setInputComment('');
      setIsComment(false);
    } catch (err) {
      console.error('Error posting comment:', err);
    }
  };

  const toolAgency = (key) => {
    switch (key) {
      case 'comment':
        setIsComment(!isComment);
        break;
      case 'like':
        handleLike();
        break;
      case 'share':
        handleShare();
        break;
      default:
        break;
    }
  };

  const handleLike = async () => {
    try {
      await axios.post('/api/article/like', { aid: id });
      if (article) {
        setArticle({
          ...article,
          likeNums: article.likeNums + 1
        });
      }
    } catch (err) {
      console.error('Error liking article:', err);
    }
  };

  const handleShare = () => {
    if (navigator.share) {
      navigator.share({
        title: article?.title,
        text: '来看看这篇文章',
        url: window.location.href
      });
    } else {
      navigator.clipboard.writeText(window.location.href);
      alert('链接已复制到剪贴板');
    }
  };

  if (loading) {
    return <div className="loading">加载中...</div>;
  }

  if (error || !article) {
    return <div className="error">{error || '文章不存在'}</div>;
  }

  const tools = {
    comment: { icon: 'comment' },
    like: { icon: 'like' },
    share: { icon: 'share' }
  };

  return (
    <div className="des-wrap">
      <div className="article-des">
        <h1 className="text-4xl mb-8">{article.title}</h1>

        <div className="article-about text-xs flex items-center mb-10">
          <span className="author-name mr-4">{article.author?.nickname}</span>
          <span className="text-common flex items-center gap-x-3">
            <span>{article.date}</span>
            <span className="flex items-center">
              <Icon name="watch" />
              <span className="ml-1">{article.clickNums}</span>
            </span>
            <span className="flex items-center">
              <Icon name="like" />
              <span className="ml-1">{article.likeNums}</span>
            </span>
          </span>
        </div>

        <article className="content markdown-body" data-theme="light" style={{ width: '100%' }} dangerouslySetInnerHTML={{ __html: article.content }}></article>

        <div>
          <div className="comment-list mt-10">
            <h1 className="mb-10 text-2xl">评论列表</h1>
            {comments.map((item) => (
              <div key={item.id} className="comment-item flex items-center gap-x-4 mb-5">
                <Avatar src={item.author?.avatar} />
                <div>
                  <div className="comment-author">{item.author?.nickname}</div>
                  <div className="comment-content">{item.content}</div>
                  <div className="comment-date">{item.date}</div>
                </div>
              </div>
            ))}
          </div>
        </div>

        {isComment && (
          <textarea 
            className="comment-submit comment fixed bottom-0" 
            onKeyPress={(e) => e.key === 'Enter' && uploadComment()}
            value={inputComment}
            onChange={(e) => setInputComment(e.target.value)}
            placeholder="输入评论发送" 
          />
        )}

      </div>

      <div className="flex flex-col items-center gap-y-4 circle-tool absolute top-2 -right-10">
        {Object.entries(tools).map(([key, value]) => (
          <button key={key} className="tool-item" onClick={() => toolAgency(key)}>
            <Icon name={value.icon} />
          </button>
        ))}
      </div>

    </div>
  );
};

export default ArticleDetail;
```

```javascript
import { defineStore } from 'pinia';
import { ref, computed } from 'vue';
import { useApi } from '../../apis';

export const useArticleStore = defineStore('article', () => {
  const articles = ref([]);
  const currentArticle = ref(null);
  const loading = ref(false);
  const error = ref(null);
  const total = ref(0);
  const page = ref(1);
  const limit = ref(10);

  const fetchArticles = async (params = {}) => {
    try {
      loading.value = true;
      error.value = null;
      const response = await useApi('articleList', {
        page: params.page || page.value,
        limit: params.limit || limit.value,
        column: params.column
      });
      articles.value = response.data;
      total.value = response.total;
      return response;
    } catch (err) {
      error.value = '获取文章列表失败';
      console.error('Error fetching articles:', err);
      return null;
    } finally {
      loading.value = false;
    }
  };

  const fetchArticleById = async (id) => {
    try {
      loading.value = true;
      error.value = null;
      const response = await useApi('articleDetail', { id });
      currentArticle.value = response;
      return response;
    } catch (err) {
      error.value = '获取文章详情失败';
      console.error('Error fetching article:', err);
      return null;
    } finally {
      loading.value = false;
    }
  };

  const createArticle = async (data) => {
    try {
      loading.value = true;
      error.value = null;
      const response = await useApi('createArticle', data);
      articles.value.unshift(response);
      return response;
    } catch (err) {
      error.value = '创建文章失败';
      console.error('Error creating article:', err);
      return null;
    } finally {
      loading.value = false;
    }
  };

  const updateArticle = async (id, data) => {
    try {
      loading.value = true;
      error.value = null;
      const response = await useApi('updateArticle', { id, ...data });
      const index = articles.value.findIndex(article => article.id === id);
      if (index !== -1) {
        articles.value[index] = response;
      }
      if (currentArticle.value && currentArticle.value.id === id) {
        currentArticle.value = response;
      }
      return response;
    } catch (err) {
      error.value = '更新文章失败';
      console.error('Error updating article:', err);
      return null;
    } finally {
      loading.value = false;
    }
  };

  const deleteArticle = async (id) => {
    try {
      loading.value = true;
      error.value = null;
      await useApi('deleteArticle', { id });
      articles.value = articles.value.filter(article => article.id !== id);
      if (currentArticle.value && currentArticle.value.id === id) {
        currentArticle.value = null;
      }
      return true;
    } catch (err) {
      error.value = '删除文章失败';
      console.error('Error deleting article:', err);
      return false;
    } finally {
      loading.value = false;
    }
  };

  const likeArticle = async (id) => {
    try {
      await useApi('likeArticle', { id });
      const article = articles.value.find(article => article.id === id);
      if (article) {
        article.likeNums += 1;
      }
      if (currentArticle.value && currentArticle.value.id === id) {
        currentArticle.value.likeNums += 1;
      }
      return true;
    } catch (err) {
      console.error('Error liking article:', err);
      return false;
    }
  };

  return {
    articles,
    currentArticle,
    loading,
    error,
    total,
    page,
    limit,
    fetchArticles,
    fetchArticleById,
    createArticle,
    updateArticle,
    deleteArticle,
    likeArticle
  };
});
```

```jsx
import { useState, useEffect, useRef } from 'react';
import axios from 'axios';
import { Avatar, Icon, Cell, Input, Button } from 'vant';

const Chat = () => {
  const [messages, setMessages] = useState([]);
  const [inputValue, setInputValue] = useState('');
  const [user, setUser] = useState(null);
  const [loading, setLoading] = useState(true);
  const messagesEndRef = useRef(null);

  useEffect(() => {
    fetchUserInfo();
    fetchMessages();
    setupWebSocket();
  }, []);

  const fetchUserInfo = async () => {
    try {
      const response = await axios.get('/api/user/info');
      setUser(response.data);
    } catch (err) {
      console.error('Error fetching user info:', err);
    }
  };

  const fetchMessages = async () => {
    try {
      const response = await axios.get('/api/chat/messages');
      setMessages(response.data);
    } catch (err) {
      console.error('Error fetching messages:', err);
    } finally {
      setLoading(false);
    }
  };

  const setupWebSocket = () => {
    const ws = new WebSocket('ws://localhost:3000');

    ws.onopen = () => {
      console.log('WebSocket connected');
    };

    ws.onmessage = (event) => {
      const message = JSON.parse(event.data);
      setMessages(prev => [...prev, message]);
    };

    ws.onclose = () => {
      console.log('WebSocket disconnected');
    };

    return () => {
      ws.close();
    };
  };

  const sendMessage = async () => {
    if (!inputValue.trim() || !user) return;

    try {
      await axios.post('/api/chat/send', {
        content: inputValue
      });
      setInputValue('');
    } catch (err) {
      console.error('Error sending message:', err);
    }
  };

  const scrollToBottom = () => {
    messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' });
  };

  useEffect(() => {
    scrollToBottom();
  }, [messages]);

  if (loading) {
    return <div className="loading">加载中...</div>;
  }

  return (
    <div className="chat-container">
      <div className="chat-header">
        <h1>聊天大厅</h1>
      </div>
      
      <div className="chat-messages">
        {messages.map((msg, index) => (
          <div 
            key={index} 
            className={`message-item ${msg.uid === user?.id ? 'own-message' : 'other-message'}`}
          >
            <Avatar src={msg.avatar} />
            <div className="message-content">
              <div className="message-author">{msg.nickname}</div>
              <div className="message-text">{msg.content}</div>
              <div className="message-time">{msg.date}</div>
            </div>
          </div>
        ))}
        <div ref={messagesEndRef} />
      </div>
      
      <div className="chat-input">
        <Input
          v-model={inputValue}
          placeholder="输入消息..."
          value={inputValue}
          onChange={(value) => setInputValue(value)}
          onKeyPress={(e) => e.key === 'Enter' && sendMessage()}
        />
        <Button type="primary" onClick={sendMessage}>发送</Button>
      </div>
    </div>
  );
};

export default Chat;
```

```javascript
const express = require('express');
const router = express.Router();
const createError = require('http-errors');
const assert = require('http-assert');
const User = require('../../models/modules/User');
const jwt = require('jsonwebtoken');
const bcrypt = require('bcrypt');

// 获取用户信息
router.get('/info', async (req, res, next) => {
  try {
    assert(req.user, '请先登录');
    
    const user = await User.findById(req.user.id).select('-password');
    assert(user, '用户不存在');
    
    res.json(user);
  } catch (err) {
    next(createError(401, err.message || '获取用户信息失败'));
  }
});

// 更新用户信息
router.put('/info', async (req, res, next) => {
  try {
    assert(req.user, '请先登录');
    
    const { nickname, avatar } = req.body;
    const user = await User.findByIdAndUpdate(req.user.id, {
      nickname,
      avatar
    }, { new: true }).select('-password');
    
    assert(user, '用户不存在');
    res.json(user);
  } catch (err) {
    next(createError(400, err.message || '更新用户信息失败'));
  }
});

// 修改密码
router.put('/password', async (req, res, next) => {
  try {
    assert(req.user, '请先登录');
    
    const { oldPassword, newPassword } = req.body;
    const user = await User.findById(req.user.id);
    
    assert(user, '用户不存在');
    
    const isMatch = await bcrypt.compare(oldPassword, user.password);
    assert(isMatch, '旧密码错误');
    
    const hashedPassword = await bcrypt.hash(newPassword, 10);
    await User.findByIdAndUpdate(req.user.id, {
      password: hashedPassword
    });
    
    res.json({ message: '密码修改成功' });
  } catch (err) {
    next(createError(400, err.message || '修改密码失败'));
  }
});
```
